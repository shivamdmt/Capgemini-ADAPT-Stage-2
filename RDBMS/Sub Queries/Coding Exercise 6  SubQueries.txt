alter session set current_schema=bank;
SET heading OFF;
SELECT ACCOUNT_NO,
TRANS_AMT
FROM BANK_SB_ACCOUNT BankSB INNER JOIN BANK_TRANSACTION Trans
on BankSB.ACCOUNT_NO = Trans.TRANS_ACC_NO
WHERE Trans.TRANS_TYPE = 'DB' AND
Trans.TRANS_AMT > (SELECT AVG(TRANS_AMT)
FROM BANK_TRANSACTION Trans
WHERE BankSB.ACCOUNT_NO = Trans.TRANS_ACC_NO);
