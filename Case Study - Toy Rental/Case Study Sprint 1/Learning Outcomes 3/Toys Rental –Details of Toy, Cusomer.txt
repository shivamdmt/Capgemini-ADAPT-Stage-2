alter session set current_schema = mt_toy;
SET heading OFF
SELECT CUST.CUSTOMER_NAME AS CUSTOMER_NAME,
TOY.TOY_NAME AS TOY_NAME,
TR.RENTAL_START_DATE,
TR.RENTAL_END_DATE
FROM CUSTOMER CUST INNER JOIN TOY_RENTAL TR
ON CUST.CUSTOMER_ID = TR.CUSTOMER_ID INNER JOIN TOY 
ON TR.TOY_ID = TOY.TOY_ID
WHERE TO_CHAR(TR.RENTAL_START_DATE,'DD-MM-YY') = '25-01-20';
