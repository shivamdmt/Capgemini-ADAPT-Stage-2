alter session set current_schema=bank;
SET heading OFF;
SELECT BANK_SB_ACCOUNT.ACCOUNT_NO AS ACCOUNT_NO,
BANK_SB_ACCOUNT.PRIMARY_CUST_ID AS PRIMARY_CUST_ID,
BANK_SB_ACCOUNT.CURR_BAL_AMT AS CURR_BAL_AMT,
BANK_SB_ACCOUNT.START_DATE AS START_DATE
FROM BANK_SB_ACCOUNT
WHERE BANK_SB_ACCOUNT.ACC_STATUS LIKE '%Active%'
ORDER BY BANK_SB_ACCOUNT.START_DATE ASC;
