alter session set current_schema = mt_toy;
SET heading OFF
SELECT CUST.CUSTOMER_NAME AS CUSTOMER_NAME,
CUST.CITY AS CITY,
TOY.RENTAL_ID AS RENTAL_ID,
TOY.RENTAL_START_DATE AS RENTAL_START_DATE,
TOY.RENTAL_END_DATE AS RENTAL_END_DATE
FROM CUSTOMER CUST LEFT OUTER JOIN
TOY_RENTAL TOY
ON CUST.CUSTOMER_ID = TOY.CUSTOMER_ID;
