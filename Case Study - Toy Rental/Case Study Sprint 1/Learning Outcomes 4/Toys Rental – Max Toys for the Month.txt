alter session set current_schema = mt_toy;
SET heading OFF
SELECT INITCAP(MONTH), TOTAL_TOYS FROM (
SELECT TO_CHAR(RENTAL_START_DATE,'MONTH') AS MONTH,
COUNT(TO_CHAR(RENTAL_START_DATE,'MONTH')) AS TOTAL_TOYS
FROM TOY_RENTAL
GROUP BY TO_CHAR(RENTAL_START_DATE,'MONTH')
ORDER BY TOTAL_TOYS DESC)
WHERE ROWNUM = 1;
