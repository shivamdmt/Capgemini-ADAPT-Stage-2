alter session set current_schema=bank;
SET heading OFF;
SELECT BANK_EMPLOYEE.DESIGNATION,
BANK_EMPLOYEE.BANK_EMP_ID,
BANK_EMPLOYEE.EMP_NAME,
BANK_EMPLOYEE.EMP_SEX
FROM BANK_EMPLOYEE
WHERE BANK_EMPLOYEE.DESIGNATION NOT LIKE '%MANAGER%'
ORDER BY BANK_EMPLOYEE.DESIGNATION DESC,
BANK_EMPLOYEE.EMP_NAME ASC;
