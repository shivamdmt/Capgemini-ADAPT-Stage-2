alter session set current_schema-bank; 
SET HEADING OFF; 
SELECT BANK_CUSTOMER.CUST_ID AS CUST_ID , 
BANK_CUST_CONTACT.CUST_PHONE AS CUST_PHONE,
BANK_CUST_CONTACT.CUST_ADDR_LINE1 AS CUST_ADDR_LINE1,
BANK_CUST_CONTACT.CUST_ADDR_LINE2 AS CUST_ADDR_LINE2,
BANK_CUST_CONTACT.CUST_CITY AS CUST_CITY,
BANK_CUST_CONTACT.CUST_STATE AS CUST_STATE,
BANK_CUST_CONTACT.CUST_PIN AS CUST_PIN
FROM BANK_CUSTOMER, BANK_CUST_CONTACT
WHERE BANK_CUSTOMER.CUST_ID = BANK_CUST_CONTACT.CUST_ID AND
BANK_CUST_CONTACT.CUST_STATE IN ( 'Karnataka' , 'Tamilnadu') AND
BANK_CUST_CONTACT.CONTACT_TYPE = 'HOME';
